## 一、`fulltext_fetcher/sources/fatcat.py`（纯逻辑 + 离线 selftest）

```python
"""Internet Archive fatcat release-lookup:按 DOI 查 IA 编目的 OA 文件全文直链。

fatcat(fatcat.wiki, Internet Archive)是 25M+ 开放获取论文的「release→file」编目;按 DOI 查
release、取其 files[].urls[](web 活链 + webarchive Wayback 直链)。DOI-keyed、零 key、只读。

为何补 unpaywall/wayback 盲区:fatcat 覆盖 IA 主动抓存的作者自存稿/机构库 OA 副本(unpaywall 常漏);
且对 URL 不可从 DOI 推导的桶(Elsevier 10.1016 / RSC 10.1039)——本仓 wayback.py 只能靠 doi.org 兜底
——fatcat 可直接给出 IA 已存档的 PDF 直链。

干净性(关键):file 由 release 的 ext_ids.doi **记录级绑定**,parse 层强制校验 ext_ids.doi==查询 DOI
(呼应经验记录 M.5:聚合源必须记录级 DOI 校验;L.5:DOI-keyed 源 0% 假阳),天然规避 websearch 假阳。

设计:纯函数解析可离线 selftest;HTTP 走 ctx.client(复用父包客户端);失败一律吞异常返 [],绝不抛
(遵 base.py 契约:单源失败不拖垮流水线)。webarchive URL 复用 wayback._to_id_url 转 id_(去工具条)。

对外接口:
    parse_files(data, want_doi) -> list[tuple[str, str]]   # [(url, kind)],记录级 DOI 校验
    find_candidate_urls(doi, cfg=None, *, fetch_json=None) -> list[tuple[str, str]]
离线自检:python -m fulltext_fetcher.sources.fatcat  → 打印 FATCAT_OK
"""
from __future__ import annotations

from typing import Any, Callable, List, Optional, Tuple

_API = "https://api.fatcat.wiki/v0/release/lookup"
_Cand = Tuple[str, str]   # (url, kind)


def _norm_doi(doi: Any) -> str:
    s = str(doi or "").strip().lower()
    for p in ("https://doi.org/", "http://doi.org/", "https://dx.doi.org/",
              "http://dx.doi.org/", "doi:"):
        if s.startswith(p):
            s = s[len(p):]
            break
    return s.strip().rstrip(".,);")


def _looks_pdf(url: str) -> bool:
    low = (url or "").lower().split("#", 1)[0].split("?", 1)[0]
    return low.endswith(".pdf") or "/pdf" in low


def _to_id_url(wb_url: str) -> str:
    """webarchive URL → id_ 原始取回形式(去 Wayback 工具条),强制 https。复用 wayback 同名逻辑。"""
    import re
    u = re.sub(r"^http://", "https://", str(wb_url or "").strip())
    if "/web/" not in u:
        return u
    if not re.search(r"/web/\d+id_/", u):
        u = re.sub(r"(/web/\d+)/", r"\1id_/", u, count=1)
    return u


def parse_files(data: Any, want_doi: str) -> List[_Cand]:
    """从 release lookup JSON 抽 [(url, kind)];**记录级校验 ext_ids.doi==want_doi**。
    只收 mimetype=application/pdf(或空)的 file;webarchive 优先(存档稳定、转 id_),web 次之。
    非 published stage 仍收(fatcat 的 file 已是实体);去重保序。绝不抛。"""
    want = _norm_doi(want_doi)
    if not want:
        return []
    try:
        got = str(((data or {}).get("ext_ids") or {}).get("doi") or "").strip().lower()
        if got != want:
            return []                                    # 记录级 DOI 校验:不符即弃(防错配)
        out: List[_Cand] = []
        seen: set = set()
        for f in (data.get("files") or []):
            mt = (f.get("mimetype") or "").lower()
            if mt and mt != "application/pdf":
                continue
            urls = f.get("urls") or []
            # webarchive 先、web 后
            ordered = ([u for u in urls if (u.get("rel") == "webarchive")]
                       + [u for u in urls if (u.get("rel") != "webarchive")])
            for u in ordered:
                url = str(u.get("url") or "").strip()
                if not url or not url.lower().startswith(("http://", "https://")):
                    continue
                if u.get("rel") == "webarchive":
                    url = _to_id_url(url)
                if url in seen:
                    continue
                seen.add(url)
                kind = "pdf" if (mt == "application/pdf" or _looks_pdf(url)) else "landing"
                out.append((url, kind))
        return out
    except Exception:  # noqa: BLE001 - 解析异常绝不外抛
        return []


def find_candidate_urls(doi: Optional[str], cfg: Any = None, *,
                        fetch_json: Optional[Callable[[str], Any]] = None) -> List[_Cand]:
    """按 DOI 查 fatcat release-lookup → [(url, kind)]。fetch_json 可注入(selftest);
    缺省用 ctx.client 由集成层传入(此处签名保 cfg 以统一)。失败吞异常返 []。"""
    d = _norm_doi(doi)
    if not d:
        return []
    fetch = fetch_json
    if fetch is None:
        return []            # 无注入取数器时不自行联网;实际取数在 free_adapters 的 FatcatSource 里用 ctx.client
    try:
        data = fetch(d)
        return parse_files(data, d)
    except Exception:  # noqa: BLE001
        return []


# ────────────────────────── 不联网 selftest ──────────────────────────
def _selftest() -> int:
    HIT = {
        "ext_ids": {"doi": "10.1088/0264-9381/19/7/380"},
        "release_stage": "published",
        "files": [
            {"mimetype": "application/pdf", "sha1": "54244fe8", "urls": [
                {"rel": "web", "url": "http://ex.org/a.pdf"},
                {"rel": "webarchive", "url": "http://web.archive.org/web/20081011163648/http://ex.org/a.pdf"},
            ]},
            {"mimetype": "application/pdf", "sha1": "954c0fb3", "urls": [
                {"rel": "webarchive", "url": "https://web.archive.org/web/20050624182645/http://ex.org/a.pdf"},
            ]},
        ],
    }
    # ① 命中:webarchive 先(转 id_)、web 后;去重
    got = parse_files(HIT, "10.1088/0264-9381/19/7/380")
    assert ("https://web.archive.org/web/20081011163648id_/http://ex.org/a.pdf", "pdf") in got, got
    assert ("http://ex.org/a.pdf", "pdf") in got, got
    assert got[0][0].endswith("id_/http://ex.org/a.pdf"), ("webarchive 应排前", got)
    assert len(got) == len(set(got)), ("去重", got)

    # ② 记录级 DOI 校验:ext_ids.doi != 查询 DOI → [](防错配,M.5)
    assert parse_files(HIT, "10.9999/other") == []
    # DOI 归一化:doi:/大小写/尾缀
    assert parse_files(HIT, "DOI:10.1088/0264-9381/19/7/380;") != []

    # ③ 非 pdf mimetype 跳过;无 files / 空 / None 优雅
    html_only = {"ext_ids": {"doi": "10.1/x"}, "files": [
        {"mimetype": "text/html", "urls": [{"rel": "web", "url": "https://ex.org/x.html"}]}]}
    assert parse_files(html_only, "10.1/x") == []
    assert parse_files({"ext_ids": {"doi": "10.1/x"}, "files": []}, "10.1/x") == []
    assert parse_files({}, "10.1/x") == [] and parse_files(None, "10.1/x") == []
    assert parse_files(HIT, "") == []

    # ④ find_candidate_urls:注入 mock 取数(HIT 含 file1 webarchive+web、file2 webarchive = 3 候选)
    res = find_candidate_urls("10.1088/0264-9381/19/7/380",
                              fetch_json=lambda d: HIT)
    assert len(res) == 3, res
    assert find_candidate_urls("", fetch_json=lambda d: HIT) == []
    assert find_candidate_urls("10.1/x", fetch_json=lambda d: (_ for _ in ()).throw(RuntimeError())) == []

    print("FATCAT_OK")
    return 0


if __name__ == "__main__":
    raise SystemExit(_selftest())
```

## 二、接线：`free_adapters.py` 加 `@register class FatcatSource`

```python
@register
class FatcatSource(BaseSource):
    """Internet Archive fatcat:按 DOI 查 IA 编目 OA 文件(web+webarchive 直链)。DOI-keyed、零 key、干净。"""
    name = "fatcat"
    requires_doi = True

    def find_candidates(self, paper: Paper, ctx: SourceContext) -> List[PdfCandidate]:
        if not paper.doi:
            return []
        try:
            from . import fatcat

            def _fetch(d: str) -> Any:
                r = ctx.client.get(fatcat._API, params={"doi": d, "expand": "files",
                                                        "hide": "abstracts,refs"})
                return r.json() if getattr(r, "status_code", 0) == 200 else None

            pairs = fatcat.find_candidate_urls(paper.doi, ctx.cfg, fetch_json=_fetch)
            # webarchive/存档直链 68、web 活链略高 72;landing 降到 45(交 landing 二抽)
            out: List[PdfCandidate] = []
            for url, kind in pairs:
                conf = (72 if kind == "pdf" and "web.archive.org" not in url
                        else 68 if kind == "pdf" else 45)
                out.append(PdfCandidate(url=url, source="fatcat", kind=kind, confidence=conf))
            return out
        except Exception:  # noqa: BLE001
            return []
```
- **注册自检**：把 `"fatcat"` 加进 `free_adapters._selftest` 的注册断言列表（现有 `("publisher_oa","oa_button","websearch","wayback","preprints","browser_search")`）。

## 三、接线位置 + confidence（DEFAULT_SOURCE_ORDER）

- **位置**：`config.DEFAULT_SOURCE_ORDER` 里排在 **DOI-keyed 干净源梯队**——`unpaywall / openalex / semantic_scholar / crossref / europe_pmc` 之后、**`wayback` 之前**（fatcat 比 wayback 的「URL 猜测」更精准，且同为 IA 存档视角，放其前先试）。
- **confidence 分级**（融进上面的排序，低于真 OA 直链、高于 landing）：
  - `web` 活链 PDF = **72**（DOI 绑定、活链）
  - `webarchive` 存档 PDF = **68**（DOI 绑定、存档稳定但可能旧版）
  - landing = **45**（交 landing.py 二抽 citation_pdf_url）
- **理由**：DOI-keyed 0 假阳（L.5），可放心排在便宜 OA 源梯队；但因是「补盲区」，排在原生 unpaywall/openalex 之后，命中靠前源即短路、不浪费请求。

## 四、SHA1 去重（download 层，非 candidate 字段）

- `PdfCandidate` **无 sha1 字段**（models.py 已核）——**别往 candidate 塞 sha1**。
- fatcat 的 `files[].sha1` 价值在 **download 层落盘后跨源去重/校验**：若实现者要用，建议在 download 记 `metadata.jsonl` 时另存已下 PDF 的 sha1，与 fatcat 声明的 sha1 比对（可选增强，非本源必需）。本源保持「只产 URL 候选」的单一职责。

- **内容 QC**：fatcat 属 DOI-keyed，download 层内容 QC 门对 DOI-keyed 源豁免（L.5）——但存档件仍建议过 `%PDF`+size 基础校验（存档可能是旧版/损坏）。

## 六、合规与边界

- 只读、零 key、DOI 校验；只取 IA 已公开存档的 OA 副本，尊重原始版权（同 wayback.py 合规口径）。
- ROI 诚实：still_missing 主体是订阅墙，fatcat 只覆盖 OA、且与 unpaywall/wayback 有重叠 → **净增点估小**；价值＝「再加一层 0 假阳 DOI-keyed 干净兜底」+ 覆盖 unpaywall 漏收的 IA 存档（尤其 elsevier/RSC 这类 wayback 只能 doi.org 兜底的桶）。
